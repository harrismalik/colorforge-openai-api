ColorForge API Documentation

Interactive documentation and testing interface for the ColorForge image processing API

GET /v1/ping
▶
Click "Test Endpoint" to check server status
CURL
curl -X GET \ "http://localhost:8000/v1/ping" \ -H "X-OpenAI-Key: YOUR_API_KEY"
JAVASCRIPT 
const response = await fetch('http://localhost:8000/v1/ping', {
  method: 'GET',
  headers: {
    'X-OpenAI-Key': 'YOUR_API_KEY'
  }
});

const data = await response.json();
console.log(data);
TYPESCRIPT 
interface PingResponse {
  status: string;
  timestamp: string;
}

const response = await fetch('http://localhost:8000/v1/ping', {
  method: 'GET',
  headers: {
    'X-OpenAI-Key': 'YOUR_API_KEY'
  }
});

const data: PingResponse = await response.json();
POST /v1/generate
▶
Configure parameters and click "Generate Image"
CURL
curl -X POST \ "http://localhost:8000/v1/generate" \ -H "Content-Type: application/json" \ -H "X-OpenAI-Key: YOUR_API_KEY" \ -d '{ "prompt": "A sunset over mountains", "size": "1024x1024", "n": 1 }'
JAVASCRIPT 
const response = await fetch('http://localhost:8000/v1/generate', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-OpenAI-Key': 'YOUR_API_KEY'
  },
  body: JSON.stringify({
    prompt: 'A sunset over mountains',
    size: '1024x1024',
    n: 1
  })
});

const data = await response.json();
console.log(data.results);
TYPESCRIPT 
interface GenerateRequest {
  prompt: string;
  size: '256x256' | '512x512' | '1024x1024';
  n: number;
}

interface GenerateResponse {
  results: Array<{ image: string }>;
}

const payload: GenerateRequest = {
  prompt: 'A sunset over mountains',
  size: '1024x1024',
  n: 1
};

const response = await fetch('http://localhost:8000/v1/generate', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-OpenAI-Key': 'YOUR_API_KEY'
  },
  body: JSON.stringify(payload)
});

const data: GenerateResponse = await response.json();

POST /v1/batch
▶
Configure batch jobs and submit
CURL
curl -X POST \ "http://localhost:8000/v1/batch" \ -H "Content-Type: application/json" \ -H "X-OpenAI-Key: YOUR_API_KEY" \ -d '{ "jobs": [ { "endpoint": "/v1/palettes", "payload": {} }, { "endpoint": "/v1/generate", "payload": { "prompt": "A serene landscape", "size": "512x512", "n": 1 } } ] }'
JAVASCRIPT
const batchJobs = { jobs: [ { endpoint: '/v1/palettes', payload: {} }, { endpoint: '/v1/generate', payload: { prompt: 'A serene landscape', size: '512x512', n: 1 } } ] }; const response = await fetch('http://localhost:8000/v1/batch', { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-OpenAI-Key': 'YOUR_API_KEY' }, body: JSON.stringify(batchJobs) }); const data = await response.json();
TYPESCRIPT
interface BatchJob { endpoint: string; payload: Record any>; } interface BatchRequest { jobs: BatchJob[]; } interface BatchResponse { batchId: string; status: string; } const batchRequest: BatchRequest = { jobs: [ { endpoint: '/v1/palettes', payload: {} }, { endpoint: '/v1/generate', payload: { prompt: 'A serene landscape', size: '512x512' as const, n: 1 } } ] }; const response = await fetch('http://localhost:8000/v1/batch', { method: 'POST', headers: { 'Content-Type': 'application/json', 'X-OpenAI-Key': 'YOUR_API_KEY' }, body: JSON.stringify(batchRequest) }); const data: BatchResponse = await response.json();
GET /v1/jobs/:jobId
▶
Enter job ID to check status
CURL
curl -X GET \ "http://localhost:8000/v1/jobs/job_abc123def456" \ -H "X-OpenAI-Key: YOUR_API_KEY"
JAVASCRIPT
const jobId = 'job_abc123def456'; const response = await fetch(`http://localhost:8000/v1/jobs/${jobId}`, { method: 'GET', headers: { 'X-OpenAI-Key': 'YOUR_API_KEY' } }); const data = await response.json(); console.log(data.status, data.results);
TYPESCRIPT
interface JobStatusResponse { jobId: string; status: 'pending' | 'processing' | 'completed' | 'failed'; results?: any[]; error?: string; } const jobId = 'job_abc123def456'; const response = await fetch(`http://localhost:8000/v1/jobs/${jobId}`, { method: 'GET', headers: { 'X-OpenAI-Key': 'YOUR_API_KEY' } }); const data: JobStatusResponse = await response.json();
